InvoiceService (المنطق الأساسي)
public class InvoiceService : IInvoiceService
{
    private readonly SmartAccountingDbContext _context;

    public InvoiceService(SmartAccountingDbContext context)
    {
        _context = context;
    }

    public async Task<InvoiceDto> CreateInvoiceAsync(CreateInvoiceDto dto)
    {
        using var transaction = await _context.Database.BeginTransactionAsync();

        var invoice = new Invoice
        {
            InvoiceNumber = $"INV-{DateTime.Now.Ticks}",
            Date = DateTime.Now,
            CustomerId = dto.CustomerId,
            Discount = dto.Discount
        };

        decimal subTotal = 0;

        foreach (var item in dto.Items)
        {
            var product = await _context.Products.FindAsync(item.ProductId);

            if (product == null)
                throw new Exception("Product not found");

            if (product.Stock < item.Quantity)
                throw new Exception("Insufficient stock");

            product.Stock -= item.Quantity;

            var lineTotal = item.Quantity * item.UnitPrice;
            subTotal += lineTotal;

            invoice.Items.Add(new InvoiceItem
            {
                ProductId = item.ProductId,
                Quantity = item.Quantity,
                UnitPrice = item.UnitPrice
            });

            // Stock Movement (important)
            _context.StockMovements.Add(new StockMovement
            {
                ProductId = item.ProductId,
                Quantity = -item.Quantity,
                Type = "SALE",
                Date = DateTime.Now
            });
        }

        invoice.SubTotal = subTotal;
        invoice.Total = subTotal - dto.Discount;

        _context.Invoices.Add(invoice);
        await _context.SaveChangesAsync();

        await transaction.CommitAsync();

        return new InvoiceDto
        {
            Id = invoice.Id,
            InvoiceNumber = invoice.InvoiceNumber,
            SubTotal = invoice.SubTotal,
            Discount = invoice.Discount,
            Total = invoice.Total
        };
    }
}

🏦 4️⃣ Treasury / Cash Flow
الدفع النقدي:
CashAccount += PaidAmount
الدفع الآجل:
AccountsReceivable += Invoice.Total
📚 5️⃣ Journal Entries Flow (أهم جزء محاسبي)
🎯 كل فاتورة = قيد محاسبي
مثال:

فاتورة بيع = 1000 جنيه نقدي

القيد:
Debit   Cash                1000
Credit  Sales Revenue       1000
لو آجل:
Debit   Accounts Receivable   1000
Credit  Sales Revenue         1000
⚠️ قاعدة ذهبية
كل Journal Entry:
Total Debit == Total Credit
